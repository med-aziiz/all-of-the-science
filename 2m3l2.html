Lesson 3.2: Software Design Patterns

Design patterns are reusable blueprints for solving common software design problems. This lesson covers three popular patterns.

1. The Singleton Pattern

Purpose: Ensure that a class has only one instance and provide a global point of access to it.

class ConfigurationManager:
    _instance = None
    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            cls._instance = super(ConfigurationManager, cls).__new__(cls)
            cls._instance.settings = {}
        return cls._instance

    def set_setting(self, key, value):
        self.settings[key] = value

    def get_setting(self, key):
        return self.settings.get(key)

# Demonstration:
config1 = ConfigurationManager()
config2 = ConfigurationManager()
config1.set_setting("theme", "dark")
print("Theme from config2:", config2.get_setting("theme"))

2. The Observer Pattern

Purpose: Establish a subscription mechanism to notify multiple objects (observers) about changes in a subject.

class Subject:
    def __init__(self):
        self._observers = []
    def attach(self, observer):
        self._observers.append(observer)
    def notify(self, message):
        for observer in self._observers:
            observer.update(message)

class Observer:
    def update(self, message):
        print("Observer received message:", message)

# Demonstration:
subject = Subject()
observer1 = Observer()
observer2 = Observer()
subject.attach(observer1)
subject.attach(observer2)
subject.notify("A new message has arrived!")

3. The Factory Pattern

Purpose: Provide a way to create objects without specifying the exact class of the object that will be created.

class Notification:
    def notify(self, message):
        pass

class EmailNotification(Notification):
    def notify(self, message):
        print("Sending Email with message:", message)

class SMSNotification(Notification):
    def notify(self, message):
        print("Sending SMS with message:", message)

class NotificationFactory:
    @staticmethod
    def create_notification(notification_type):
        if notification_type == "email":
            return EmailNotification()
        elif notification_type == "sms":
            return SMSNotification()
        else:
            raise ValueError("Unknown notification type")

# Demonstration:
notification = NotificationFactory.create_notification("email")
notification.notify("Welcome to our service!")
